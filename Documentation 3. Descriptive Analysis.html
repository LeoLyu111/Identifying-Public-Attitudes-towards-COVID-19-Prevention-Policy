Function data_clean() is used to delete the repeated weibowhich some are repeated because of unexpanded, called ""

In [4]:
def data_clean(data_name): # 
    for i in range(data_name.shape[0]):
        weibo_day = data_name.loc[i].dropna()
        for j in range(len(weibo_day)-1):
            if weibo_day[j][:5] == weibo_day[j+1][:5]: # This judgement is to judge whether the weibo is repeated. As the weibo only 
                                                       # repeat is adjacent, while the 1st one is unexpended and the 2nd one is what we
                                                       # need. We regard the first five words repeated as the same weibo.
                weibo_day[j] = np.nan   #set Nan to the previous unexpanded repeated weibo.
            data_name.loc[i] = weibo_day
    return data_name

citys = ['Beijing','Guangzhou','Shanghai','Shenzhen']

def creat_txt(city,weibo):
    f = open(city+"_.txt",'a') # creat a txt file for different city 
    for i in range(len(weibo)): # go through all Weibo in each day
        daily_wb = weibo.iloc[i].dropna() # drop nan(the repeated Weibo we deleted before)
        for j in range(len(daily_wb)): 
            sent_wb = daily_wb[j] # go through each Weibo in one day
            f.write(sent_wb) # save every Weibo into txt file
            f.write('\n')
    f.close() # close txt file
    return f

def word_cloud(data_file,image): # data_file: Weibo and policies text data
                                 # image: word clouds background
    fn = open(data_file) # open the txt file created before
    string_data = fn.read() 
    fn.close() 

    # Text pre-processing
    pattern = re.compile(u'\t|\n|\.|-|:|;|\)|\(|\?|"') 
    string_data = re.sub(pattern, '', string_data) 

    # Text Splitting
    StopWords = '.txt'  # dictionary of stop words 
                               # for example: ['','','','','','','','']
    seg_list_exact = jieba.lcut(string_data, cut_all = False) # use jieba doing Text Splitting
    object_list = []

    with open(StopWords, 'r') as meaninglessFile:
        stopwords = set(meaninglessFile.read().split('\n'))
    stopwords.add(' ')
    for word in seg_list_exact:         
        if word not in stopwords:       
            object_list.append(word)  # we do not want to calculate the stop words  


    # Statistics on word frequency
    word_counts = collections.Counter(object_list) # Statistics on word frequency
    word_counts_top10 = word_counts.most_common(10) # top 10 most frequent words
    print (word_counts_top10) 
    
    # draw bar plot for top 10 most frequent words
    word = []
    freq = [] 
    for l in word_counts_top10:
        word.append(l[0])
        freq.append(l[1])
    word.reverse() # just reverse let the plot show from most frequent to less frequent words
    freq.reverse()
    matplotlib.rc('font', family='Hiragino Sans GB', weight='bold', size=15) #family is used to show Chinese words
    plt.rcParams['axes.unicode_minus'] = False
    fig = plt.figure(figsize=(10,6))
    plt.title(data_file[:-7])
    plt.barh(word,freq)
    plt.show()

    # Word frequency display
    mask = np.array(Image.open(image)) # Define word frequency Background 
    wc = wordcloud.WordCloud(
        background_color='black', # set the color of background 
        font_path='/System/Library/Fonts/Hiragino Sans GB.ttc', # Set font format
        mask=mask, # Set background image
        max_words=200, # Maximum number of words displayed
        max_font_size=100 , # Font Max
        scale=32  # Adjust the picture clarity, the larger the value the clearer
    )

    wc.generate_from_frequencies(word_counts) # Generating word clouds from dictionaries
    image_colors = wordcloud.ImageColorGenerator(mask) # Create a color scheme from the background image
    wc.recolor(color_func=image_colors) # Set the word cloud color to the background image scheme
    wc.to_file(data_file[:-7]+"_.jpg") # Exporting images to files
    plt.imshow(wc) # Show word cloud
    plt.axis('off') # Close axis
    plt.show() # Display image
